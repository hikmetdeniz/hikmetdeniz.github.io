    // Accordion: same section => only one open
    function wireAccordions(){
      document.querySelectorAll('.accordion[data-accordion="one"]').forEach(acc => {
        const items = Array.from(acc.querySelectorAll('.acc-item'));

        function setState(item, open){
          item.classList.toggle('active', open);
          const s = item.querySelector('.state');
          if(s) s.textContent = open ? 'Kapat' : 'Aç';
        }

        items.forEach(item => {
          const head = item.querySelector('.acc-head');
          const toggle = () => {
            const isOpen = item.classList.contains('active');
            items.forEach(it => setState(it, false));
            setState(item, !isOpen);
          };

          head.addEventListener('click', toggle);
          head.addEventListener('keydown', (e)=>{
            if(e.key === 'Enter' || e.key === ' ') { e.preventDefault(); toggle(); }
          });

          // init
          setState(item, false);
        });
      });
    }

    wireAccordions();
    document.getElementById('y').textContent = new Date().getFullYear();
